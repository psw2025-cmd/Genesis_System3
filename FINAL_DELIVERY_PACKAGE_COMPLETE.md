```python
# FUNCTION 2: Detect drift for each model
def detect_drift_for_underlying(
    underlying: str,
    model_metrics: Dict[str, Any]
) -> Dict[str, Any]:
    """
    Check 3 drift conditions:
    1. Accuracy < 55% (ACCURACY_THRESHOLD)
    2. Test samples < 10 (MIN_TEST_SAMPLES)
    3. Precision/Recall imbalance (ratio < 0.3)
    
    Return PROMOTE or REJECT decision.
    """
    accuracy = model_metrics.get('accuracy')
    precision = model_metrics.get('precision', 0.0)
    recall = model_metrics.get('recall', 0.0)
    test_samples = model_metrics.get('test_samples', 0)
    
    drift_detected = False
    reasons = []
    
    # Condition 1: Low accuracy
    if accuracy < 0.55:
        drift_detected = True
        reasons.append(f"Low accuracy: {accuracy:.1%} < 55%")
    
    # Condition 2: Insufficient samples
    if test_samples < 10:
        drift_detected = True
        reasons.append(f"Insufficient samples: {test_samples} < 10")
    
    # Condition 3: Imbalance
    if precision > 0 and recall > 0:
        ratio = min(recall/precision, precision/recall)
        if ratio < 0.3:
            drift_detected = True
            reasons.append(f"Precision/Recall imbalance")
    
    decision = "REJECT" if drift_detected else "PROMOTE"
    
    return {
        "drift_detected": drift_detected,
        "reasons": reasons,
        "decision": decision
    }
```

```python
def read_latest_evaluation_metrics(
    eval_dir: str = "logs",
    pattern: str = "phase249_model_evaluation_*.json",
    min_required_keys: Optional[List[str]] = None
) -> Optional[Dict[str, Any]]:
    """
    Read the latest Phase 249 Extended evaluation JSON.
    
    This function finds the most recent evaluation report generated by
    evaluate_phase249_models.py and returns its contents safely.
    
    Args:
        eval_dir: Directory containing evaluation JSON files
        pattern: Glob pattern to match evaluation files
        min_required_keys: Required keys in the JSON
    
    Returns:
        Dict with evaluation metrics, or None if no valid file found.
        
        Structure:
        {
            'evaluation_timestamp': '2025-12-06T14:30:45.123456',
            'total_models': 5,
            'models': {
                'NIFTY': {
                    'status': 'SUCCESS',
                    'accuracy': 0.462,
                    'precision': 0.0,
                    'recall': 0.0,
                    'f1_score': 0.0,
                    'test_samples': 13,
                    'training_accuracy': 0.875
                },
                'BANKNIFTY': {...},
                'FINNIFTY': {...},
                'MIDCPNIFTY': {...},
                'SENSEX': {...}
            },
            'summary': {
                'evaluated_models': 5,
                'avg_accuracy': 0.431,
                'min_accuracy': 0.308,
                'max_accuracy': 0.462
            }
        }
    """
    # Set default required keys
    if min_required_keys is None:
        min_required_keys = ['models', 'evaluation_timestamp']
    
    # Resolve directory path
    if not Path(eval_dir).is_absolute():
        eval_path = PROJECT_ROOT / eval_dir
    else:
        eval_path = Path(eval_dir)
    
    # Ensure directory exists
    if not eval_path.exists():
        logger.warning(f"[LSTM_UTILS] Evaluation directory not found: {eval_path}")
        return None
    
    try:
        # Find all matching evaluation files
        search_pattern = str(eval_path / pattern)
        matching_files = sorted(glob.glob(search_pattern))
        
        if not matching_files:
            logger.warning(f"[LSTM_UTILS] No evaluation files found matching: {search_pattern}")
            return None
        
        # Get the most recent file (sorted, so last one is newest)
        latest_file = matching_files[-1]
        logger.info(f"[LSTM_UTILS] Reading latest evaluation: {latest_file}")
        
        # Load JSON file
        with open(latest_file, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        # Validate required keys exist
        missing_keys = [key for key in min_required_keys if key not in data]
        if missing_keys:
            logger.warning(f"[LSTM_UTILS] Missing required keys: {missing_keys}")
            return None
        
        # Validate models key is a dict
        if not isinstance(data.get('models'), dict):
            logger.warning(f"[LSTM_UTILS] 'models' key is not a dictionary")
            return None
        
        logger.info(f"[LSTM_UTILS] Successfully loaded evaluation metrics for {len(data['models'])} models")
        return data  # ✅ Returns Dict with all evaluation data
    
    except json.JSONDecodeError as e:
        logger.warning(f"[LSTM_UTILS] Failed to parse JSON: {e}")
        return None  # ✅ Never raises, returns None
    
    except Exception as e:
        logger.warning(f"[LSTM_UTILS] Unexpected error: {e}")
        return None  # ✅ Never raises, always graceful
```

**Key Feature:** Safe error handling - returns Dict | None, never crashes
